local lfs = require "lfs"

function getProjectPath()
  local path = lfs.currentdir():gsub("\\", "/")
  local llmk = io.open(path.."/llmk.toml", "r")

  while llmk == nil do
    path = string.match(path, "(.+)/")
    llmk = io.open(path.."/llmk.toml", "r")
  end

  io.close(llmk)
  tex.sprint(path)
end
\end{luacode*}
